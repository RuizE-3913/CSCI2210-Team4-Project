class Airline {
- airlineName : String
- flights : ArrayList<Flight> 
- staff : ArrayList<Staff> 
- aircraft : ArrayList<Aircraft>
- airports : ArrayList<Airport>

+ Airline(String)
+ addFlight(Flight) : void
+ addAircraft(Aircraft) : void
+ hireStaff(Staff) : void
}

class Airport { 
- airportCode : String 
- name : String  
- location : String 
- terminals  : ArrayList<Terminal>

+ Airport(String, String)
+ getAirportInfo() : String
}

class Terminal {
 - terminalId: String
 - name: String
 - gates : ArrayList<String>
  
+ Terminal(String, String)
+ getTerminalInfo() : String
}

class Aircraft { 
- aircraftID : String  
- model : String   
- capacity : int
- status : String 
- seats : ArrayList<Seat>

+ Aircraft(String, String, int)
+ getSeats() : ArrayList<Seat>
+ updateStatus(String) : void
}

class Flight {
- flightNumber: String
- origin: Airport
- destination: Airport
- departureTerminal: Terminal
- arrivalTerminal: Terminal
- departureTime: Schedule
- arrivalTime: Schedule
- status: String
- aircraft: Aircraft

+ Flight (String, Airport, Airport, Terminal, Terminal)
+ getAvailableSeats (): int
+ updateStatus (String): void
}

class Schedule {
- scheduleID : String
- departureTime : String
- arrivalTime : String
- flights : ArrayList<Flight>

+ Schedule(String, Date)
+ addFlight(Flight) : void
+ removeFlight(Flight) : void
+ getFlights() : ArrayList<Flight>
}

class Passenger { 
- passengerID: String
- name : String
- passportNumber : String
- contactInfo : String

+ Passenger (String, String)
+ updateContactInfo (String) : void
+ viewTickets () : ArrayList<Ticket>
}

class Booking {
- bookingID : String
- bookingDate : Date 
- status : String 
- tickets : ArrayList<Ticket>

+ Booking(String)
+ addTicket(Ticket) : void
+ cancelBooking() : void
+ getTickets() : ArrayList<Ticket>
}

class Payment {
  - paymentId: String
  - booking: Booking
  - amount: double
  - paymentDate: Date
  - method: String	// e.g., "credit_card", "paypal", "bank_transfer"
  - status: String	// e.g., "PENDING", "COMPLETED", "FAILED", "REFUNDED"
  
  + Payment(Booking booking, double amount, String method)
  + processPayment(): boolean	// returns success or failure
  + getPaymentId(): String
  + getAmount(): double
  + getPaymentDate(): Date
  + getMethod(): String
  + getStatus(): String
  + updateStatus(String newStatus): void	// for refunds, etc.
  + isCompleted(): boolean
}

class Ticket { 
- ticketID : String

- seat: Seat
- status : String 

+ Ticket(String, Flight)
+ assignSeat(Seat) : void 
+ cancelTicket() : void
+ getTicketDetails() : String
}

class Seat { 
- seatNumber : String
- classType : String

+ Seat(String, String)
+ isSeatAvailable() : boolean
}

class Staff {
- staffID : String

- contactInfo : String

+ updateContactInfo (String) : void

}

class Baggage { 
- baggageID : String

- status : String  

+ Baggage(String, double)
+ updateStatus(String) : void
}

class StaffAssignment {
- assignmentID : String

+ assignCrewMember(Staff) : void
+ removeCrewMember(Staff) : void
